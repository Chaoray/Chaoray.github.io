class TextBox {
    string text;

    TextBox addText(const string& append) {
        text += append;
        return this;
    }
};

TextBox textBox1;
textBox1.addText("Text1").addText("Text2").addText("Text3");

class TextBox {
    string text;

    TextBox& addText(const string& append) {
        text += append;
        return *this;
    }
};

TextBox textBox1;
textBox1.addText("Text1").addText("Text2").addText("Text3");

Works fine!

class TextBox {
    string text;

    TextBox* addText(const string& append) {
        text += append;
        return this;
    }
};

TextBox textBox1;
(*(*textBox1.addText("Text1")).addText("Text2")).addText("Text3");

Longer, more verbose.
