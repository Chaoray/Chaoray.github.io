// events\guildCreate.js

const { Events, } = require('discord.js');

module.exports = {
    name: Events.GuildCreate,
    async execute(guild) {
        guild.commands.set(guild.client.slashCommands.map((command) => command.data));
    },
};

// events\guildDelete.js
module.exports = {
    name: Events.GuildDelete,
    async execute(guild) {
        guild.commands.set([]);
    },
};
